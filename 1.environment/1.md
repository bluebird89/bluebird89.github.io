# Devlopment Setup #
## Vagrant ##
1. config
    Vagrant.configure("2") do |config|
    		config.vm.box = "ubuntu/trusty64"
    		config.vm.network "forwarded_port", guest: 80, host: 5000
    		config.vm.synced_folder "../", "/vagrant"
    end
2. vagrant up
